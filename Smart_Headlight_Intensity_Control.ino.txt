// Smart Headlight Intensity Control System
// Components: Arduino UNO, LDR sensor, LED (as headlight)

#define LDR A0
#define LED 9

int lightValue;

void setup() {
  pinMode(LED, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  lightValue = analogRead(LDR); // Read ambient light
  int brightness = map(lightValue, 0, 1023, 255, 0); 
  // Brighter when road is dark, dim when light is high

  analogWrite(LED, brightness);

  Serial.print("Light Value: ");
  Serial.print(lightValue);
  Serial.print(" | LED Brightness: ");
  Serial.println(brightness);

  delay(200);
}
